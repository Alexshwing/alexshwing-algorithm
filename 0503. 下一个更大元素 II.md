```python
class Solution:
    def nextGreaterElements(self, A: List[int]) -> List[int]:
        n = len(A)
        A = A + A
        ans = [-1] * n
        stk = []
        for i in range(2 * n - 1, -1, -1):
            while stk and A[stk[-1]] <= A[i]: stk.pop()
            if stk and i < n: ans[i] = A[stk[-1]]
            stk.append(i)
        return ans
```
