```python
class Node:
    __slot__ = 'son', 'cnt'

    def __init__(self):
        self.son = dict() # å­˜ pair , (s[0], s[-1]), (s[1], s[-2])...
        self.cnt = 0

class Solution:
    def countPrefixSuffixPairs(self, A: List[str]) -> int:
        ans = 0
        root = Node()
        for s in A:
            cur = root
            for p in zip(s, reversed(s)):
                if p not in cur.son:
                    cur.son[p] = Node()
                cur = cur.son[p]
                ans += cur.cnt
            cur.cnt += 1
        return ans
```
