class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        n = len(nums)
        dp = [0] * (n + 1)
        sz = 0
        for i, x in enumerate(nums):
            l, r = 0, sz
            while l < r:
                mid = (l + r + 1) // 2
                if dp[mid] < x: l = mid
                else: r = mid - 1
            sz = max(sz, r + 1)
            dp[r + 1] = x
        return sz
