class LinkNode:
    # 提高访问属性的速度，并节省内存
    __slot__ = 'key', 'value', 'freq', 'prev', 'next'

    def __init__(self, key=-1, value=-1):
        self.key = key
        self.value = value
        self.freq = 1
        self.prev = None
        self.next = None

class DoublyLinkedList:
    def __init__(self):
        self.head = LinkNode(-1, -1)
        self.tail = LinkNode(-1, -1)
        self.head.next = self.tail
        self.tail.prev = self.head

    def add_first(self, node: LinkNode):
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node
    
    def remove(self, node: LinkNode):
        node.next.prev = node.prev
        node.prev.next = node.next
        node.next, node.prev = None, None
        return node
    
    def remove_last(self):
        return self.remove(self.tail.prev)
    
    def is_empty(self) -> bool:
        return self.head.next == self.tail

class LFUCache:

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.min_freq = 0
        self.map = defaultdict(LinkNode) # 值 -> 节点
        self.freq_map = defaultdict(DoublyLinkedList) # 频率 -> 双向链表

    def get(self, key: int) -> int:
        if self.capacity == 0 or key not in self.map:
            return -1
        node = self.map[key]
        self.incr_freq(node)
        return node.value

    def put(self, key: int, value: int) -> None:
        if self.capacity == 0:
            return
        if key in self.map:
            node = self.map[key]
            node.value = value
            self.incr_freq(node)
            return
        if len(self.map) == self.capacity:
            ls = self.freq_map[self.min_freq]
            node = ls.remove_last()
            self.map.pop(node.key)
        node = LinkNode(key, value)
        self.add_node(node)
        self.map[key] = node
        self.min_freq = 1

    def incr_freq(self, node: LinkNode):
        freq = node.freq
        ls = self.freq_map[freq]
        ls.remove(node)
        if ls.is_empty():
            self.freq_map.pop(freq)
            if freq == self.min_freq:
                self.min_freq += 1
        node.freq += 1
        self.add_node(node)

    def add_node(self, node: LinkNode):
        freq = node.freq
        ls = self.freq_map[freq]
        ls.add_first(node)
        self.freq_map[freq] = ls
