class Solution:
    def getAverages(self, A: List[int], k: int) -> List[int]:
        w = 2 * k + 1 # çª—å£å¤§å°
        ans = [-1] * len(A)
        c = 0
        for i, x in enumerate(A):
            c += x
            if i >= w - 1:
                ans[i - k] = c // w
                c -= A[i - w + 1]
        return ans
