class Solution:
    def maxSlidingWindow(self, A: List[int], k: int) -> List[int]:
        res = []
        dq = Deque()
        for i, x in enumerate(A):
            while len(dq) and i - dq[0] >= k: dq.popleft() # 左边收缩保证返回
            while len(dq) and A[i] >= A[dq[-1]]: dq.pop() # 右边弹出无用
            dq.append(i)
            if i >= k - 1: res.append(A[dq[0]])
        return res
