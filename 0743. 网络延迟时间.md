class Solution:
    def networkDelayTime(self, times: List[List[int]], n: int, k: int) -> int:
        g = [[] for _ in range(n)]
        for x, y, w in times:
            g[x - 1].append((y - 1, w))

        def dijkstra(start: int = 0, end: int = n - 1) -> List[int]:
            dist = [inf] * n
            dist[start] = 0
            pq = [(0, start)]
            while pq:
                dis, x = heappop(pq)
                if dis > dist[x]: continue
                for y, w in g[x]:
                    if dist[y] > dist[x] + w:
                        dist[y] = dist[x] + w
                        heappush(pq, (dist[y], y))
            return dist

        dist = dijkstra(k - 1)
        ans = max(dist)
        return ans if ans != inf else -1
